32 float LTC2448IC::takeSample(int8_t channel) {
33  if(channel > 15 || channel < 0) {

35  }

37  uint8_t futureChannel = oracle(channel);
38 
39  if(channel == lastChannel) {
40  // on oracle hit, we only do one transaction
41  waitForSampleComplete();
42  makeTransaction(futureChannel, OSR64, buf);
43  }
44  else {
45  makeTransaction(channel, OSR64, buf);
46  // on oracle miss, we need to perform multiple transactions
47  waitForSampleComplete();
48  makeTransaction(futureChannel, OSR64, buf);
49  }

51  // decode the LTC2448 response
52  uint8_t msByte = buf[0]; // most significant byte
53  uint32_t reading = ((buf[0] & 0x1F) << 24) + (buf[1] << 16) + (buf[2] << 8) + buf[3];
54  reading = reading >> 5; // bottom 5 bits are garbage
55  bool busy = (reading & 0x80) != 0;
56  if(busy) {
57  throw runtime_error("LTC2448 is still busy! (check timing constraints)");
58  }
59  bool negative = (reading & 0x20) != 0;
60 
61  float volts = (5.0 * reading) / (1 << 24);
62  if(negative) {
63  volts *= -1;
64  }
65  return volts;
66 }
